4. 恢复与验证方法

4.1 恢复单个文件

# 以 mission.h 为例
cd ~/oh/foundation/ability/ability_runtime/services/abilitymgr/include/mission/
cp mission.h.bk mission.h

4.2 查看修改差异

# 对比备份与当前文件的差异
diff mission.h.bk mission.h

4.3 批量恢复所有文件

#!/bin/bash
# restore_all_backups.sh
OH=~/oh

files=(
  "foundation/ability/ability_runtime/interfaces/inner_api/ability_manager/include/ability_manager_interface.h"
  "foundation/ability/ability_runtime/services/abilitymgr/include/mission/mission.h"
  "foundation/ability/ability_runtime/services/abilitymgr/src/mission/mission.cpp"
  "foundation/ability/ability_runtime/services/appmgr/include/remote_client_manager.h"
  "foundation/ability/ability_runtime/services/appmgr/src/app_mgr_service_inner.cpp"
  "foundation/bundlemanager/bundle_framework/interfaces/inner_api/appexecfwk_base/include/bundle_constants.h"
  "foundation/bundlemanager/bundle_framework/interfaces/inner_api/appexecfwk_core/src/bundlemgr/bundle_mgr_host.cpp"
  "foundation/bundlemanager/bundle_framework/services/bundlemgr/include/ipc/installd_interface.h"
  "foundation/bundlemanager/bundle_framework/services/bundlemgr/src/bundle_installer.cpp"
  "foundation/graphic/graphic_2d/rosen/modules/render_service_base/src/feature/buffer_reclaim/rs_buffer_reclaim.cpp"
  "third_party/musl/arch/aarch64/bits/syscall.h.in"
  "third_party/musl/src/internal/ADLTSection.h"
  "build/config/components/ets_frontend/ets2abc_config.gni"
  "build/ohos/app/app_internal.gni"
  "foundation/arkui/ui_lite/ext/updater/BUILD.gn"
  "third_party/libjpeg-turbo/BUILD.gn"
)

for f in "${files[@]}"; do
  full="$OH/$f"
  if [ -f "${full}.bk" ]; then
    cp "${full}.bk" "$full"
    echo "RESTORED: $f"
  else
    echo "SKIP (no backup): $f"
  fi
done

5. 注意事项

repo sync 会覆盖补丁：执行 repo sync 后，OH 源文件会被还原为上游版本， 此时 .bk 备份文件也会变得多余。但 补丁必须在下次编译前重新应用。
备份内容：所有 .bk 文件保存的是 修改前的原始文件， 即 OH 源码 weekly_20260302 的未修改版本。可用于与当前文件对比，确认补丁内容。
路径映射差异：本项目 ohos_patches/ 中的目录结构与 OH 源码树的实际路径存在部分差异：
  • ohos_patches/ability_rt/ → ~/oh/foundation/ability/ability_runtime/
  • ohos_patches/bundle_framework/ → ~/oh/foundation/bundlemanager/bundle_framework/
  • ohos_patches/graphic_2d/ → ~/oh/foundation/graphic/graphic_2d/
  • ohos_patches/build/ → ~/oh/build/ 及 ~/oh/foundation/、~/oh/third_party/
  • installd_interface.h：patch 目录标注为 include/installd/，实际为 include/ipc/
